\section{Motivation}

\begin{frame}{Generic Drug Quality: A Regulated Industry With Visibility Gaps}
\begin{columns}[T]
\column{0.52\textwidth}
\begin{itemize}
  \setlength{\itemsep}{8pt}
  \item 90\% of U.S. prescriptions are generic drugs
  \item FDA inspects each manufacturing facility roughly every 1--2 years
  \item After each visit, inspectors issue a \textbf{Form 483}: a written record of every manufacturing violation observed
  \item Between visits, the 483 sits unread -- no quality signal reaches regulators while patients receive drugs
\end{itemize}

\begin{frame}{The Form 483: Public, Detailed, and Unread}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{What is a Form 483?}
\begin{itemize}
  \item Issued at the close of every FDA facility inspection
  \item Lists every manufacturing violation observed (typically 3--15 per visit)
  \item Topics include contamination, data integrity, lab controls, and quality systems
  \item Filed publicly within days of the inspection -- no standard schema
\end{itemize}

\column{0.47\textwidth}
\begin{block}{Example observation (paraphrased)}
\small
``Investigations into out-of-specification results are not conducted with a scientifically sound approach. Five OOS potency results were invalidated without adequate justification\ldots''
\end{block}

\vspace{8pt}
{\small Plain regulatory prose, written fresh by each inspector. No structured database. Filed after every inspection -- but the text has never been systematically used to predict future quality failures.}
\end{columns}
\end{frame}

\begin{frame}{FDA Inspection Outcomes and the Targeting Gap}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Three inspection outcomes:}
\begin{itemize}
  \item \textcolor{govGreen}{\textbf{NAI}} -- No Action Indicated
  \item \textcolor{warnOrange}{\textbf{VAI}} -- Voluntary Action Indicated
  \item \textcolor{ncsuRed}{\textbf{OAI}} -- Official Action Indicated: FDA mandates corrective action
\end{itemize}

\vspace{8pt}
\textbf{FDA's Site Selection Model (SSM)} scores each facility using:
\begin{itemize}
  \item Compliance history (NAI/VAI/OAI outcome)
  \item Recalls and field alert reports
  \item Patient exposure and inherent product risk
  \item Time since last inspection
\end{itemize}

\vspace{4pt}
\textbf{Not in SSM:} the content of the 483 -- what violations were cited, how severe, how widespread

\column{0.45\textwidth}
\textbf{The gap this creates:}
\begin{itemize}
  \item Most VAI facilities receive the same SSM score -- 483 text content not considered
  \item 483 text is available \emph{immediately after inspection}, before recalls or AEs surface
  \item Undetected quality problems can escalate to recalls and supply disruptions -- one pathway to drug shortage
\end{itemize}

\vspace{6pt}
\begin{alertblock}{Research questions}
\small
\begin{enumerate}
  \item Do 483 text signals predict serious adverse events?
  \item Among VAI facilities -- where SSM provides no differentiation -- does text separate high- from low-harm producers?
\end{enumerate}
\end{alertblock}
\end{columns}
\end{frame}

\section{LLM Extraction Pipeline}

\begin{frame}{Step 00: What a 483 Observation Looks Like in Practice}
{\small \textbf{Sun Pharma, Mohali -- Aug 2022 OAI inspection.} 8 observations across 8 pages. Two shown below.}

\vspace{5pt}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Observation 1 \hfill \textcolor{warnOrange}{\scriptsize Data Integrity}}
\scriptsize
\textbf{Header:} ``There is a failure to thoroughly review any unexplained discrepancy whether or not the batch has been already distributed.''

\vspace{4pt}
\textbf{Sub-item:} ``Disclosure 2020-01 confirmed instances of backdating by QA and QC personnel. The investigation was not thorough to evaluate the scope of backdating in the QA and QC departments.''

\vspace{3pt}
\textit{[3 sub-items with nested detail -- 2 pages]}
\end{block}

\column{0.48\textwidth}
\begin{block}{Observation 2 \hfill \textcolor{techBlue}{\scriptsize Lab Controls}}
\scriptsize
\textbf{Header:} ``Established sampling plans, test procedures and laboratory control mechanisms are not documented at the time of performance.''

\vspace{4pt}
\textbf{Sub-item:} ``Building access records showed an employee responsible for collecting samples did not enter the buildings where samples were documented to have been collected.''

\vspace{3pt}
\textit{[2 sub-items with dated examples -- 1.5 pages]}
\end{block}
\end{columns}

\vspace{6pt}
{\small \textbf{Same inspection. Two different domains.} Domain, severity, and patient risk are all implicit in the prose -- no structured fields, no shared schema. This is what the LLM reads.}
\end{frame}

\begin{frame}{Step 01: What the LLM Extracts and Why}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Five layers per observation:}

\vspace{4pt}
\begin{itemize}
  \item \textbf{L1: Domain.} 8 CFR\,211 areas: Lab Controls, Quality System, Production, Buildings \& Equipment, Materials, Records, Packaging, Personnel.
  \item \textbf{L2: Severity.} Confirmed defect vs.\ systemic risk. Major means bad product was likely already made.
  \item \textbf{L3: Flags.} DI, contamination, patient risk, scope, root cause, remediation. Each independently defined.
  \item \textbf{L4: Recurrence.} Persistent findings signal a facility that hasn't fixed the problem.
  \item \textbf{L5: Co-occurrence.} Lab Controls and DI together carries more signal than either alone.
\end{itemize}

\column{0.47\textwidth}
\textbf{Expert review (Redica, June 2026):}

\vspace{5pt}
\begin{itemize}
  \setlength{\itemsep}{4pt}
  \item Observation-level is the right unit of analysis
  \item Severity rule aligns with how regulators distinguish enforceability
  \item Capital vs.\ Cultural root cause split is well-grounded in GMP practice
  \item $\sim$80\% no remediation signal is expected -- 483s predate facility responses
\end{itemize}

\vspace{7pt}
\textit{Limitations (future work):}
\begin{itemize}
  \setlength{\itemsep}{4pt}
  \item DI binary flag collapses deliberate falsification and procedural misses
  \item Contamination flag under-captures non-sterile products
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Step 01: Turning Unstructured Inspection Text into Structured Features}
\begin{columns}[T]
\column{0.58\textwidth}
\begin{block}{Input: verbatim 483 observation text}
\scriptsize
``Batch production records were not completed at the time of manufacture. Entries were reconstructed from memory 2 days after processing.''
\end{block}

\vspace{4pt}
\textbf{Output (Claude Haiku, one pass per observation):}

\vspace{2pt}
{\scriptsize
\begin{tabular}{lll}
\toprule
Dimension & Extracted value & Rationale \\
\midrule
Severity & \textbf{Major} & confirmed defect; records reconstructed \\
DI flag & \textbf{True} & reconstructed entries = DI trigger \\
Scope & \textbf{SingleBatch} & failure tied to one product run \\
Root cause & \textbf{Cultural} & behavioral / management gap \\
Remediation & \textbf{None} & no corrective action language \\
\bottomrule
\end{tabular}
}

\column{0.39\textwidth}
\textbf{How trustworthy is the extraction?}

\vspace{5pt}
\begin{itemize}
  \setlength{\itemsep}{5pt}
  \item Prompt definitions reviewed by Redica GMP specialists (June 2026) -- key dimensions align with industry practice
  \item Convergent validity against Redica's independent classifications: severity 67.5\%, domain 58.6\%
  \item Output is a fixed JSON schema -- the model cannot add or invent fields
  \item Independent human labeling of a random sample is in progress; results here should be treated as preliminary
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Step 02: From Observation Labels to Inspection Feature Vectors}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Three types of features per inspection:}

\vspace{4pt}
\textbf{Type 1: Share} (normalized by observation count):
\begin{itemize}
  \item \texttt{vc\_labcontrols\_share}: fraction of observations in Lab Controls domain
  \item Captures what kind of inspection this was
\end{itemize}

\vspace{4pt}
\textbf{Type 2: Count} (absolute intensity):
\begin{itemize}
  \item \texttt{n\_labcontrols\_obs}: raw number of Lab Controls observations
  \item Captures how much was cited, not just whether it was cited at all
\end{itemize}

\vspace{4pt}
\textbf{Type 3: Joint co-occurrence}:
\begin{itemize}
  \item \texttt{joint\_labctrl\_di}: 1 if both Lab Controls and DI appear in the same inspection
  \item Strong predictor of future AEs ($\rho = 0.26$ at Q+4, $p < .01$; see slide 13)
\end{itemize}

\column{0.45\textwidth}
\textbf{Why co-occurrence is the strongest signal:}

\vspace{4pt}
{\small
Lab Controls alone: a QC process failed.\\[2pt]
DI alone: a record was altered.\\[2pt]
\textbf{Both together:} conditions associated with defective product and compromised documentation appear in the same inspection. The correlation with future AEs is stronger when both co-occur than when either appears alone.
}

\vspace{8pt}
\begin{block}{Output}
\small 246 inspections $\times$ 17 features.\\
This matrix feeds the correlation analysis and the prediction model directly.
\end{block}

\vspace{4pt}
{\scriptsize Share and count are complementary: share is robust to long inspections; count captures absolute burden. Both enter the model.}
\end{columns}
\end{frame}

\begin{frame}{From 483 Text to Future Harm: Two Mechanisms}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Why does 483 text predict next-year AEs?}

\vspace{8pt}
\textbf{OAI facilities}
\begin{itemize}
  \item Product made before the visit is already distributed
  \item 483 text describes the quality conditions present at that time
  \item AEs emerge as inventory clears and enforcement begins
\end{itemize}

\vspace{6pt}
\textbf{VAI facilities}
\begin{itemize}
  \item No corrective action mandated
  \item 483 text describes the facility's quality state at the time of inspection
  \item Those conditions persist through the post-inspection period
\end{itemize}

\vspace{6pt}
{\small Both patterns run in the same direction: more severe 483 text is associated with higher AEs in Q+1..Q+4.}

\column{0.45\textwidth}
\textbf{How we operationalize it:}

\vspace{6pt}
\begin{tabular}{ll}

\midrule
Unit & One inspection event \\
Features & 483 text at that inspection \\
N & 246 inspections, 98 FEIs \\
CV & FEI-grouped 5-fold \\
\bottomrule
\end{tabular}

\vspace{8pt}
\textbf{Question:} for the 73 VAI-only facilities where FDA's outcome label never varies, does 483 text still separate high- from low-harm producers?
\end{columns}

\end{itemize}

\textbf{Lab Controls, not data integrity, is associated with elevated AEs one year later.}
\end{columns}

% ── SECTION 6: PREDICTIVE MODEL ──────────────────────────────────────────────
\section{Predictive Model}

\begin{frame}{Prediction Setup}
\begin{columns}[T]
\column{0.52\textwidth}
\begin{itemize}
  \item \textbf{Unit:} One inspection event
  \item \textbf{Features:} \tech{17 LLM text signals} from the 483 at that inspection
  \item \textbf{Outcome:} Above-median serious AEs in the post-inspection year
        $$\mathbf{1}\!\left[\,\textstyle\sum_{q=1}^{4}\,\text{AE}_{t+q} > \text{median}\,\right]$$
        {\scriptsize OAI inspections: 33 of 246 (13\%)}
  \item \textbf{CV:} FEI-grouped 5-fold
  \item \textbf{Model:} Logistic regression; random forest as robustness check
\end{itemize}

\column{0.45\textwidth}
\begin{alertblock}{AUC = 0.5 is the floor}
The outcome is 50/50 by construction. A random classifier scores exactly 0.5.

\vspace{4pt}
Does 483 text beat that floor -- and does it beat the OAI label alone?
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{483 Text Best Predicts Harm in Facilities FDA Never Penalized}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{AUC, 5-fold FEI-grouped CV (246 inspections, manufacturer-specific AEs):}

\vspace{4pt}
{\small
\begin{tabular}{clc}
\toprule
 & Feature set & LR AUC \\
\midrule
A & Text only (17 LLM features) & 0.503 \\
B & Text + OAI flag & 0.527 \\
C & OAI flag only & 0.539 \\
\midrule
\multicolumn{2}{l}{\textit{Random baseline}} & \textit{0.500} \\
\midrule
\highlight{D} & \highlight{Text, VAI-only facilities} & \highlight{0.586} \\
E & Text, OAI-ever facilities & 0.537 \\
\bottomrule
\end{tabular}
}

\vspace{6pt}
{\small Config D: 73 VAI-only FEIs, 165 inspections. FDA never classified any inspection as OAI at these facilities -- the OAI flag has no variation, so text is the only predictor available.}

\column{0.47\textwidth}
{\small
\textbf{Full population (A vs C):} Text and the OAI flag perform similarly near random. Neither is a strong standalone classifier.

\vspace{5pt}
\textbf{VAI-only (D):} Text reaches \highlight{AUC\,=\,0.59} -- the strongest result. Among facilities FDA never formally penalized, 483 text is the only signal that separates high- and low-harm producers.

\vspace{5pt}
\textbf{OAI-ever (E):} Text is no better in facilities that did receive enforcement. The enforcement action itself may have resolved the quality problem, flattening future AEs.

\vspace{5pt}
\textbf{Takeaway:} 483 text is most valuable precisely where structured regulatory signals are absent.
}
\end{columns}
\end{frame}

% ── SECTION 10: SUMMARY ──────────────────────────────────────────────────────
\section{Summary}

\begin{frame}{Summary: Five Key Messages}

\begin{enumerate}
  \setlength{\itemsep}{6pt}
  \item \textbf{483 text predicts harm where structured signals are absent.}\\
    In facilities FDA never formally penalized (VAI-only), text-only LR AUC = \highlight{0.586}. In the full population, text and the OAI flag are near random and similar to each other.

  \item \textbf{Lab Controls -- not data integrity -- is the leading signal.}\\
    Four Lab Controls features reach significance at Q+4 ($\rho$ up to 0.28***). Data integrity shows no significant association at any lag.

  \item \textbf{Inspection is reactive by design; 483 text is not.}\\
    FDA's Site Selection Model uses AEs and recalls to rank facilities. Rising AEs before inspection are partly expected. 483 text from the prior visit contains information the SSM does not incorporate.

  \item \textbf{Same VAI label, very different outcomes.}\\
    Hi-sig VAI facilities have 63\% more AEs at Q+4 than Lo-sig VAI (49 vs.\ 30). FDA's outcome label assigns them all the same category.

  \item \textbf{Strategic leniency candidates are identifiable.}\\
    Facilities with repeated high Lab Controls signals and no OAI history can be ranked. Cross-referencing with market concentration will test the shortage-pressure hypothesis.
\end{enumerate}

% ── SECTION 11: LIMITATIONS & NEXT STEPS ─────────────────────────────────────
\section{Limitations \& Next Steps}

\begin{frame}{Limitations}
\begin{columns}[T]
\column{0.50\textwidth}
\begin{itemize}
  \item \textbf{Sample size:} 98 FEIs, 33 OAI inspections. Subgroup results (VAI-only AUC\,=\,0.59) are hypothesis-generating, not confirmatory.
  \item \textbf{COVID gap:} 2020--21 moratorium left 4-year inter-inspection windows; post-2022 baselines are confounded.
  \item \textbf{FAERS coverage:} manufacturer-specific attribution matches 47.5\% of serious AEs. Unmatched reports are from manufacturers outside our sample.
\end{itemize}

\column{0.47\textwidth}
\begin{itemize}
  \item \textbf{Shared ANDAs:} one ANDA can cover multiple facilities. AE counts are proportional to market share, not individual facility output.
  \item \textbf{Extraction quality:} LLM labels are validated against one reference (Redica) that reads the same 483 text -- neither is inspector ground truth. Two specific weaknesses: the contamination flag is primarily tuned to sterile and injectable products, so non-sterile cases may be under-captured; the DI binary flag collapses deliberate falsification and procedural misses into one signal (partially mitigated by capturing severity separately).
  \item \textbf{Volume:} commercial volume not integrated; facility size is a potential confounder.
  \item \textbf{Causal:} all associations are correlational. Strategic leniency interpretation requires market concentration data to test the shortage-pressure hypothesis.
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Next Steps}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Near-term:}
\begin{enumerate}
  \item Add market volume as covariate to rule out confounding by facility size
  \item Robustness check excluding 2020--21 COVID inspection gap
  \item Test whether results hold within individual APIs
  \item Cross-check leniency candidates with recall and import refusal records
\end{enumerate}

\column{0.47\textwidth}
\textbf{Future work:}
\begin{enumerate}
  \item Commercial sales data for volume and market concentration
  \item Extend outcome to drug shortage
  \item Refine LLM extraction: extend contamination to non-sterile products, align DI coding with ALCOA gradations, validate with independent human labeling
  \item Real-time facility risk dashboard post-inspection
\end{enumerate}

\vspace{6pt}
\begin{block}{}
\small Target: full paper submission; AAAI 2027 (AI/policy angle)
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Backup: FAERS Filtering — Design Rationale}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Why Primary Suspect only?}\\[4pt]
FAERS reports list all drugs a patient was taking. Only the PS was judged by the reporter to have caused the event. Including concomitant or interacting drugs would attribute AEs to bystander medications and inflate counts at every facility.

\vspace{8pt}
\textbf{Why first-word name match?}\\[4pt]
``Atorvastatin Calcium'' in \texttt{prod\_ai} has two words. Any-word matching recovers a handful of missed combination products but mis-attributes ``Calcium'' rows to Calcium Gluconate FEIs. First-word-only is the safer tradeoff; the unmatched 0.9\% is negligible.

\column{0.47\textwidth}
\textbf{Why not a rate (AEs / volume)?}\\[4pt]
SDUD Medicaid volume covers only 69\% of FEI-years. Dividing by a partially missing denominator introduces systematic bias — facilities with missing SDUD look artificially high-risk. AE count enters as the raw outcome; $\log(\text{SDUD units})$ enters as a covariate when available.

\vspace{8pt}
\textbf{Why serious outcomes only?}\\[4pt]
Non-serious reports are subject to much higher underreporting variability. Serious AEs have mandatory reporting paths (MedWatch, expedited), making the count more comparable across facilities and time periods.
\end{columns}
\end{frame}
