% ── SECTION 1: MOTIVATION ────────────────────────────────────────────────────
\section{Motivation}

\begin{frame}{The FDA 483 Form: Rich Signal, Rarely Used}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{What is a Form 483?}
\begin{itemize}
  \item Issued at the end of every FDA facility inspection
  \item Lists \emph{observations}: specific cGMP violations cited during the visit
  \item Each observation is 1–3 paragraphs of regulatory prose
  \item Topics range from contamination and data integrity to lab controls, equipment, and quality system failures
\end{itemize}
\vspace{4pt}
\textbf{The problem}: 483s are public, detailed, and filed immediately after every inspection, yet they remain \emph{unstructured text} with no standardized schema across facilities or inspectors.

\column{0.42\textwidth}
\begin{block}{Example 483 observation (paraphrased)}
\small
``Investigations into out-of-specification results are not conducted with scientific sound and thorough approach. Specifically, the investigation for Batch \#XY failed to extend to other batches manufactured under similar conditions\ldots''
\end{block}
\vspace{6pt}
\begin{alertblock}{Research question}
Can LLM extraction of 483 text predict \textbf{facility-level patient harm} (serious adverse events) before FDA's next inspection?
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{Why This Matters}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Manufacturing quality failures drive two interconnected risks:}
\begin{enumerate}
  \item \textbf{Patient harm:} bad product already distributed causes adverse events before FDA detects the problem
  \item \textbf{Supply disruption:} quality problems may trigger voluntary recalls, production shutdowns, or import refusals; though OAI outcomes often lead to corrective action rather than shortage, concentrated markets amplify any capacity loss
\end{enumerate}

\vspace{8pt}
\textbf{The core problem:} FDA inspections happen every 1--2 years on average. Between visits, no structured quality signal reaches regulators. If 483 text can predict harm, that window can be shortened.

\column{0.42\textwidth}
\begin{block}{What this enables}
\small
\begin{itemize}
  \item Prioritize re-inspection for facilities whose 483 text shows AE-linked signals, before the next scheduled visit
  \item Flag VAI/NAI outcomes where Lab Controls and DI text indicate the classification understates actual quality risk
  \item Provide a risk signal between inspections: when a facility's last 483 is 3--4 years old, text is the only available input
\end{itemize}
\end{block}
\end{columns}
\end{frame}

% ── SECTION 4: LLM EXTRACTION ────────────────────────────────────────────────
\section{LLM Extraction Pipeline}

\begin{frame}{Step 00: What Raw 483 Text Looks Like}
\begin{columns}[T]
\column{0.56\textwidth}
\textbf{Two observations from the same 2022 inspection (paraphrased):}

\vspace{5pt}
{\small
\textbf{Obs.\ 1.} ``Batch production records were not completed at the time of manufacture. Entries were reconstructed from memory 2 days after processing.''

\vspace{5pt}
\textbf{Obs.\ 2.} ``Investigations into out-of-specification results are not conducted with a scientifically sound and thorough approach. Five OOS potency results were invalidated without adequate scientific justification.''
}

\vspace{8pt}
{\footnotesize One inspection may have 3--15 observations. No standard schema across inspectors or facilities. Domain, severity, and root cause are all implicit in the prose.}

\column{0.41\textwidth}
\textbf{What we need to extract:}

\vspace{4pt}
{\small
\begin{tabular}{ll}
\toprule
Question & Example answer \\
\midrule
Domain? & Lab Controls, DI \\
Severity? & Major (bad product made) \\
Patient risk? & Yes \\
Data integrity? & Yes (reconstructed) \\
Scope? & Facility-wide \\
\bottomrule
\end{tabular}
}

\vspace{8pt}
Step 01 runs Claude Haiku on each observation and outputs a strict JSON covering all five dimensions. 98 FEIs, 246 inspections, 1,067 observations total.
\end{columns}
\end{frame}

\begin{frame}{Step 01: What the LLM Extracts and Why}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Five layers per observation:}

\vspace{4pt}
\begin{itemize}
  \item \textbf{L1: Domain.} 8 CFR\,211 areas: Lab Controls, Quality System, Production, Buildings \& Equipment, Materials, Records, Packaging, Personnel.
  \item \textbf{L2: Severity.} Confirmed defect vs.\ systemic risk. Major means bad product was likely already made.
  \item \textbf{L3: Flags.} DI, contamination, patient risk, scope, root cause, remediation. Each independently defined.
  \item \textbf{L4: Recurrence.} Persistent findings signal a facility that hasn't fixed the problem.
  \item \textbf{L5: Co-occurrence.} Lab Controls and DI together carries more signal than either alone.
\end{itemize}

\column{0.47\textwidth}
\textbf{Expert endorsement (Redica, June 2026):}

\vspace{4pt}
Redica's GMP specialists reviewed our prompt definitions and confirmed:
\begin{itemize}
  \item Observation-level is the right unit of analysis
  \item DI binary flag fits supply risk prediction
  \item Repeat and patient risk are ``obviously high-impact''
  \item Our domain philosophy is ``not wrong''
\end{itemize}

\vspace{6pt}
Step 02 rolls these observation-level labels into one row per inspection. That feature vector goes directly into the correlation analysis and the prediction model.
\end{columns}
\end{frame}

\begin{frame}{Step 01: How the LLM Extracts Signals, a Concrete Example}
\begin{columns}[T]
\column{0.58\textwidth}
\begin{block}{Input: verbatim 483 observation text}
\scriptsize
``Batch production records were not completed at the time of manufacture. Entries were reconstructed from memory 2 days after processing.''
\end{block}

\vspace{4pt}
\textbf{Output: Claude Haiku, strict JSON prompt:}

\vspace{2pt}
{\scriptsize
\begin{tabular}{llp{3.6cm}}
\toprule
Dimension & Value & Prompt rule (excerpt) \\
\midrule
Severity & \textbf{Major} & actual defect confirmed\ldots batch records reconstructed \\
DI flag & \textbf{True} & backdating, deleted records, reconstructed entries \\
Scope & \textbf{SingleBatch} & failure tied to one batch or one product run \\
Root cause & \textbf{Cultural} & behavioral, procedural, or management gap \\
Remediation & \textbf{None} & no corrective action language present \\
Investigation & \textbf{False} & True only if investigation was not performed \\
\bottomrule
\end{tabular}
}

\column{0.39\textwidth}
\textbf{Prompt design:}
\begin{itemize}
  \item Strict JSON schema: the model cannot hallucinate fields
  \item Each of the 10 dimensions has its own rule and anchor examples
  \item All 10 extracted in one pass per observation
  \item Model: Claude Haiku (cost-efficient at scale)
\end{itemize}

\vspace{6pt}
{\scriptsize Independent SME labeling of a random sample is in progress.}
\end{columns}
\end{frame}

\begin{frame}{Step 02: From Observation Labels to Inspection Feature Vectors}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Three types of features per inspection:}

\vspace{4pt}
\textbf{Type 1: Share} (normalized by observation count):
\begin{itemize}
  \item \texttt{vc\_labcontrols\_share}: fraction of observations in Lab Controls domain
  \item Captures what kind of inspection this was
\end{itemize}

\vspace{4pt}
\textbf{Type 2: Count} (absolute intensity):
\begin{itemize}
  \item \texttt{n\_labcontrols\_obs}: raw number of Lab Controls observations
  \item Captures how much was cited, not just whether it was cited at all
\end{itemize}

\vspace{4pt}
\textbf{Type 3: Joint co-occurrence}:
\begin{itemize}
  \item \texttt{joint\_labctrl\_di}: 1 if both Lab Controls and DI appear in the same inspection
  \item Highest predictor of future AEs ($\rho = 0.22$, $p < .01$)
\end{itemize}

\column{0.45\textwidth}
\textbf{Why co-occurrence is the strongest signal:}

\vspace{4pt}
{\small
Lab Controls alone: a QC process failed.\\[2pt]
DI alone: a record was altered.\\[2pt]
\textbf{Both together:} bad product was likely made and the evidence was manipulated. The probability that subpotent or contaminated product reached patients is higher when both are present.
}

\vspace{8pt}
\begin{block}{Output}
\small 246 inspections $\times$ 17 features.\\
This matrix feeds the correlation analysis and the prediction model directly.
\end{block}

\vspace{4pt}
{\scriptsize Share and count are complementary: share is robust to long inspections; count captures absolute burden. Both enter the model.}
\end{columns}
\end{frame}

% ── FAERS DETAIL SLIDE ────────────────────────────────────────────────────────

\begin{frame}{How We Construct the FAERS AE Signal}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Four-step processing funnel:}

\vspace{4pt}
\begin{enumerate}
  \item \textbf{ANDA filter} (parquet build): keep generic drug reports only. Branded reports are excluded; our FEIs are ANDA facilities.
  \item \textbf{Primary Suspect only} (\texttt{role\_cod = PS}): the reporter named this drug as the primary cause. Concomitant and interacting drugs are dropped.
  \item \textbf{Serious outcomes only}: keep reports where the worst outcome is Death, Hospitalization, Life-threatening, Disability, Congenital anomaly, Required intervention, or Other serious. ``No outcome reported'' is excluded (14.6\% of rows, $\approx$24,700 reports dropped).
  \item \textbf{Name matching}: first word of \texttt{prod\_ai}, lowercased $\to$ \texttt{api\_key}. Inner join to the Valisure FEI map (14 APIs $\times$ 129 FEIs). Match rate: \highlight{99.1\%}.
\end{enumerate}

\column{0.47\textwidth}
\textbf{Design rationale:}

\vspace{6pt}
\textbf{Why PS only?} FAERS reports list all drugs a patient was taking. Only the Primary Suspect was judged by the reporter to have caused the event. Including concomitant drugs would attribute AEs to bystander medications.

\vspace{6pt}
\textbf{Why first-word match?} ``Atorvastatin Calcium'' in \texttt{prod\_ai} has two words. Any-word matching recovers a handful of missed combination products but mis-attributes ``Calcium'' rows to Calcium Gluconate FEIs. First-word-only is the safer tradeoff; the unmatched 0.9\% is negligible.

\vspace{6pt}
\textbf{Why not a rate?} SDUD volume covers only 69\% of FEI-years. Dividing by a partially missing denominator introduces systematic bias. AE count enters as the raw outcome; $\log(\text{SDUD units})$ enters as a covariate.
\end{columns}
\end{frame}

% ── SECTION 6: PREDICTIVE MODEL ──────────────────────────────────────────────
\section{Predictive Model}

\begin{frame}{Panel Construction and Outcome Variable}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Unit of observation:} one inspection event.\\[4pt]
Features come from the 483 text at that inspection. Outcome is measured in the 4 quarters after.

\vspace{6pt}
\textbf{Outcome:} Binary \texttt{ae\_high\_next4q}
$$\mathbf{1}\!\left[\sum_{q=1}^{4} n\_ae_{t+q} > \text{median}\right]$$
Sum of serious FAERS reports in Q+1 through Q+4 after inspection, via DailyMed NDC$\to$FEI crosswalk.

\vspace{6pt}
\textbf{Cross-validation:} FEI-grouped 5-fold. All inspections of the same facility stay in one fold — no facility-level leakage.

\vspace{6pt}
\textbf{Models:} Logistic Regression (L2, $C=1$) and Random Forest (300 trees, max depth 4).

\column{0.45\textwidth}
\textbf{Panel statistics:}\\[4pt]
\begin{tabular}{ll}
\toprule
Metric & Value \\
\midrule
Inspection events & 246 \\
Unique FEIs & 98 \\
Outcome base rate & 50.0\% \\
OAI inspections & 26 (10.6\%) \\
\bottomrule
\end{tabular}

\vspace{6pt}
\textbf{AE window ($\pm$4 quarters around inspection):}\\[4pt]
\begin{tabular}{ll}
\toprule
Column & Meaning \\
\midrule
\texttt{n\_ae\_tm4} & AEs 4 qtrs before (1 yr) \\
\texttt{n\_ae\_tm2} & AEs 2 qtrs before (6 mo) \\
\texttt{n\_ae\_t0}  & AEs at inspection quarter \\
\texttt{n\_ae\_tp2} & AEs 2 qtrs after (6 mo) \\
\texttt{n\_ae\_tp4} & AEs 4 qtrs after — outcome \\
\bottomrule
\end{tabular}
\end{columns}
\end{frame}

\begin{frame}{Model Results: Ablation by Feature Set}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{FEI-grouped 5-fold CV (mean AUC, 246 inspections):}

\vspace{4pt}
\begin{tabular}{llcc}
\toprule
Config & Features & LR AUC & RF AUC \\
\midrule
A & \highlight{Text only} (17 LLM) & \highlight{0.528} & 0.502 \\
B & Text + OAI flag & 0.531 & 0.499 \\
C & OAI flag only & 0.518 & 0.518 \\
\midrule
\multicolumn{4}{l}{\textit{Stratified subgroups}} \\
D & Text, VAI-only FEIs & 0.513 & 0.494 \\

\midrule
\multicolumn{4}{l}{\textit{Random baseline = 0.500}} \\
\bottomrule
\end{tabular}

\vspace{6pt}
\begin{itemize}
  \item Text-only LR beats the OAI flag baseline by \highlight{+1.0 AUC points} (0.528 vs 0.518). The gap is modest; the trajectory analysis (Hi-sig VAI persist = 1.10) shows stronger discrimination than the supervised model at this sample size.
  \item Adding OAI flag to text (config B, 0.531) gives a marginal boost over text alone. The label carries some signal but not much beyond what text already captures.
  \item RF underperforms LR throughout: 246 rows, the ensemble overfits.
\end{itemize}

\column{0.42\textwidth}
\textbf{Config D vs E:}

\begin{itemize}
  \item \textbf{Config D (VAI-only, 79 FEIs, 182 rows)}: AUC = 0.513 — near random. For facilities FDA never escalated, text has no predictive power on average.
  \item \textbf{Config E (OAI-ever, 19 FEIs, 64 rows)}: LR AUC = 0.525. With 19 FEIs in a 5-fold CV, results are unstable. Treat as exploratory; the trajectory analysis is a more reliable signal.
\end{itemize}

\vspace{6pt}
\begin{block}{Implication}
Text-only model marginally outperforms the OAI-flag baseline (+1 AUC pt). The trajectory evidence is stronger: Hi-sig VAI facilities show more sustained harm than OAI facilities, identified by 483 text alone.
\end{block}
\end{columns}
\end{frame}

% ── SECTION 7: INSPECTION GAP ANALYSIS ───────────────────────────────────────
\section{Inspection Gap Analysis}

\begin{frame}{Inspection Gaps Shape What the Trajectory Means}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{AE trajectory by gap length} (inspection-centered quarterly panel, 237 matched inspections):

\vspace{4pt}

\midrule

\textbf{The critical case: 1–2yr gap group}\\
$t_{-4}$ is within the current inspection cycle (no overlap with previous inspection). This group shows the clearest pre-distribution signal: AEs \textbf{+27\%} in the year before inspection.

\column{0.45\textwidth}
\textbf{Why gaps matter for interpretation:}
\begin{itemize}
  \item For gap $>$2yr: our $t_{-4}$ point (1 year before current inspection) is approximately at the time of the \emph{previous inspection}. The pre-rise comparison is against a post-inspection baseline, not a neutral one.
  \item For gap $<$1yr: follow-up inspections; already-elevated AEs from the prior inspection cycle persist.
  \item For gap $>$3.5yr: AEs are essentially flat post-inspection (persist = 1.033). Long inter-inspection windows may reflect facilities with voluntary self-corrections already underway, or lower baseline risk.
\end{itemize}

\vspace{4pt}
\begin{alertblock}{Refined claim}
``AEs rise 27\% in the year \emph{before} inspection for annually-inspected facilities (1--2yr gap). FDA is detecting an ongoing quality problem with a $\sim$1 year lag.''
\end{alertblock}
\end{columns}
\end{frame}

\section{Limitations \& Next Steps}

\begin{frame}{Limitations}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Sample size:}
\begin{itemize}
  \item 98 FEIs, 246 inspections, only \textbf{26 OAI inspections}
  \item Subgroup analyses (Config D, E, clusters) are underpowered
  \item Treat cluster findings as hypothesis-generating, not confirmatory
\end{itemize}

\vspace{6pt}
\textbf{COVID inspection gap:}
\begin{itemize}
  \item 2020–2021: only 26 inspections (moratorium)
  \item Post-2022 inspections have 4-year prior gap — the $t_{-2}$ baseline is in COVID, a very different environment
  \item AE trends during COVID are confounded by healthcare utilization changes
\end{itemize}

\vspace{6pt}
\textbf{FAERS reporting bias:}
\begin{itemize}
  \item Serious AEs only; underreporting is non-random
  \item Reporting rates vary by drug class, year, and market events
  \item No adjustment for reporting trends
\end{itemize}

\column{0.47\textwidth}
\textbf{LLM extraction circularity:}
\begin{itemize}
  \item LLM validated against Redica, which uses the same 483 text. Both are human-coded; neither is inspector ground truth.
  \item True external validation requires FDA inspector notes or a direct clinical outcome linkage.
\end{itemize}

\vspace{6pt}
\textbf{SDUD volume:}
\begin{itemize}
  \item 31\% of FEI-years have no SDUD match, biasing the AE rate denominator
  \item Medicaid covers $\sim$20\% of the market; commercial volume (IQVIA) is not yet integrated
\end{itemize}

\vspace{6pt}
\textbf{Causal identification:}
\begin{itemize}
  \item All associations are correlational; unobserved confounders (R\&D investment, country of origin, API source) are not controlled
  \item The strategic leniency interpretation rests on market concentration data we do not yet have
\end{itemize}
\end{columns}
\end{frame}

% ── SECTION 12: SUMMARY ──────────────────────────────────────────────────────
\section{Summary}

\begin{frame}{Summary: Five Key Messages}

\begin{enumerate}
  \setlength{\itemsep}{8pt}
  \item \textbf{LLM text signals carry AE-predictive information beyond the OAI flag.}\\
    Text-only LR AUC = \highlight{0.528} vs 0.518 OAI-flag baseline (+1.0 point). The supervised model result is modest at 246 rows. Trajectory evidence is stronger: Hi-sig VAI facilities (identified by 483 text alone) show AE persistence of 1.10 vs 1.03 for OAI-ever facilities.

  \item \textbf{Five text signals carry AE-predictive information; twelve do not.}\\
    Lab Controls count, DI flag, and their joint co-occurrence predict AEs one year out. Quality System share, severity tier, and cultural root cause do not. The split tracks which failures leave bad product in the market.

  \item \textbf{Inspection is a lagging indicator.}\\
    AEs peak \emph{at} the inspection year, rising 1–2 years before FDA arrives. For facilities inspected every 1–2 years, AEs rise +27\% in the 4 quarters before inspection.

  \item \textbf{483 text separates high-risk VAI facilities that classification misses.}\\
    Hi-signal VAI (top-quartile Lab Controls + DI text): AEs at Q+4 are 10\% above inspection level (persist = 1.10) and 34\% above 1 year prior. OAI-ever facilities, by contrast, show partial correction (persist = 1.02). FDA rated both groups the same — text identified the worse outcome.

  \item \textbf{Strategic leniency candidates are identifiable.}\\
    3 FEIs show rising AEs, high technical 483 signals, and zero OAI ever. Cross-referencing with market concentration will test the shortage-avoidance hypothesis.
\end{enumerate}
