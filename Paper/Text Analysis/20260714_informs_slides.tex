% ── SECTION 1: MOTIVATION ────────────────────────────────────────────────────
\section{Motivation}

\begin{frame}{The FDA 483 Form: Rich Signal, Rarely Used}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{What is a Form 483?}
\begin{itemize}
  \item Issued at the end of every FDA facility inspection
  \item Lists \emph{observations}: specific cGMP violations cited during the visit
  \item Each observation is 1–3 paragraphs of regulatory prose
  \item Topics range from contamination and data integrity to lab controls, equipment, and quality system failures
\end{itemize}
\vspace{4pt}
\textbf{The problem}: 483s are public, detailed, and filed immediately after every inspection, yet they remain \emph{unstructured text} with no standardized schema across facilities or inspectors.

\column{0.42\textwidth}
\begin{block}{Example 483 observation (paraphrased)}
\small
``Investigations into out-of-specification results are not conducted with scientific sound and thorough approach. Specifically, the investigation for Batch \#XY failed to extend to other batches manufactured under similar conditions\ldots''
\end{block}
\vspace{6pt}
\begin{alertblock}{Research question}
Can LLM extraction of 483 text predict \textbf{facility-level patient harm} (serious adverse events) before FDA's next inspection?
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{Why This Matters}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Manufacturing quality failures drive two interconnected risks:}
\begin{enumerate}
  \item \textbf{Patient harm:} bad product already distributed causes adverse events before FDA detects the problem
  \item \textbf{Supply disruption:} quality problems may trigger voluntary recalls, production shutdowns, or import refusals; though OAI outcomes often lead to corrective action rather than shortage, concentrated markets amplify any capacity loss
\end{enumerate}

\vspace{8pt}
\textbf{The core problem:} FDA inspections happen every 1--2 years on average. Between visits, no structured quality signal reaches regulators. If 483 text can predict harm, that window can be shortened.

\column{0.42\textwidth}
\begin{block}{What this enables}
\small
\begin{itemize}
  \item Prioritize re-inspection for facilities whose 483 text shows AE-linked signals, before the next scheduled visit
  \item Flag VAI/NAI outcomes where Lab Controls and DI text indicate the classification understates actual quality risk
  \item Provide a risk signal between inspections: when a facility's last 483 is 3--4 years old, text is the only available input
\end{itemize}
\end{block}
\end{columns}
\end{frame}

% ── SECTION 4: LLM EXTRACTION ────────────────────────────────────────────────
\section{LLM Extraction Pipeline}

\begin{frame}{Step 00: What Raw 483 Text Looks Like}
\begin{columns}[T]
\column{0.56\textwidth}
\textbf{Two observations from the same 2022 inspection (paraphrased):}

\vspace{5pt}
{\small
\textbf{Obs.\ 1.} ``Batch production records were not completed at the time of manufacture. Entries were reconstructed from memory 2 days after processing.''

\vspace{5pt}
\textbf{Obs.\ 2.} ``Investigations into out-of-specification results are not conducted with a scientifically sound and thorough approach. Five OOS potency results were invalidated without adequate scientific justification.''
}

\vspace{8pt}
{\footnotesize One inspection may have 3--15 observations. No standard schema across inspectors or facilities. Domain, severity, and root cause are all implicit in the prose.}

\column{0.41\textwidth}
\textbf{What we need to extract:}

\vspace{4pt}
{\small
\begin{tabular}{ll}
\toprule
Question & Example answer \\
\midrule
Domain? & Lab Controls, DI \\
Severity? & Major (bad product made) \\
Patient risk? & Yes \\
Data integrity? & Yes (reconstructed) \\
Scope? & Facility-wide \\
\bottomrule
\end{tabular}
}

\vspace{8pt}
Step 01 runs Claude Haiku on each observation and outputs a strict JSON covering all five dimensions. 98 FEIs, 246 inspections, 1,067 observations total.
\end{columns}
\end{frame}

\begin{frame}{Step 01: What the LLM Extracts and Why}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Five layers per observation:}

\vspace{4pt}
\begin{itemize}
  \item \textbf{L1: Domain.} 8 CFR\,211 areas: Lab Controls, Quality System, Production, Buildings \& Equipment, Materials, Records, Packaging, Personnel.
  \item \textbf{L2: Severity.} Confirmed defect vs.\ systemic risk. Major means bad product was likely already made.
  \item \textbf{L3: Flags.} DI, contamination, patient risk, scope, root cause, remediation. Each independently defined.
  \item \textbf{L4: Recurrence.} Persistent findings signal a facility that hasn't fixed the problem.
  \item \textbf{L5: Co-occurrence.} Lab Controls and DI together carries more signal than either alone.
\end{itemize}

\column{0.47\textwidth}
\textbf{Expert endorsement (Redica, June 2026):}

\vspace{4pt}
Redica's GMP specialists reviewed our prompt definitions and confirmed:
\begin{itemize}
  \item Observation-level is the right unit of analysis
  \item DI binary flag fits supply risk prediction
  \item Repeat and patient risk are ``obviously high-impact''
  \item Our domain philosophy is ``not wrong''
\end{itemize}

\vspace{6pt}
Step 02 rolls these observation-level labels into one row per inspection. That feature vector goes directly into the correlation analysis and the prediction model.
\end{columns}
\end{frame}

\begin{frame}{Step 01: How the LLM Extracts Signals, a Concrete Example}
\begin{columns}[T]
\column{0.58\textwidth}
\begin{block}{Input: verbatim 483 observation text}
\scriptsize
``Batch production records were not completed at the time of manufacture. Entries were reconstructed from memory 2 days after processing.''
\end{block}

\vspace{4pt}
\textbf{Output: Claude Haiku, strict JSON prompt:}

\vspace{2pt}
{\scriptsize
\begin{tabular}{llp{3.6cm}}
\toprule
Dimension & Value & Prompt rule (excerpt) \\
\midrule
Severity & \textbf{Major} & actual defect confirmed\ldots batch records reconstructed \\
DI flag & \textbf{True} & backdating, deleted records, reconstructed entries \\
Scope & \textbf{SingleBatch} & failure tied to one batch or one product run \\
Root cause & \textbf{Cultural} & behavioral, procedural, or management gap \\
Remediation & \textbf{None} & no corrective action language present \\
Investigation & \textbf{False} & True only if investigation was not performed \\
\bottomrule
\end{tabular}
}

\column{0.39\textwidth}
\textbf{Prompt design:}
\begin{itemize}
  \item Strict JSON schema: the model cannot hallucinate fields
  \item Each of the 10 dimensions has its own rule and anchor examples
  \item All 10 extracted in one pass per observation
  \item Model: Claude Haiku (cost-efficient at scale)
\end{itemize}

\vspace{6pt}
{\scriptsize Independent SME labeling of a random sample is in progress.}
\end{columns}
\end{frame}

\begin{frame}{Step 02: From Observation Labels to Inspection Feature Vectors}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Three types of features per inspection:}

\vspace{4pt}
\textbf{Type 1: Share} (normalized by observation count):
\begin{itemize}
  \item \texttt{vc\_labcontrols\_share}: fraction of observations in Lab Controls domain
  \item Captures what kind of inspection this was
\end{itemize}

\vspace{4pt}
\textbf{Type 2: Count} (absolute intensity):
\begin{itemize}
  \item \texttt{n\_labcontrols\_obs}: raw number of Lab Controls observations
  \item Captures how much was cited, not just whether it was cited at all
\end{itemize}

\vspace{4pt}
\textbf{Type 3: Joint co-occurrence}:
\begin{itemize}
  \item \texttt{joint\_labctrl\_di}: 1 if both Lab Controls and DI appear in the same inspection
  \item Strong predictor of future AEs ($\rho = 0.26$ at Q+4, $p < .01$; see slide 13)
\end{itemize}

\column{0.45\textwidth}
\textbf{Why co-occurrence is the strongest signal:}

\vspace{4pt}
{\small
Lab Controls alone: a QC process failed.\\[2pt]
DI alone: a record was altered.\\[2pt]
\textbf{Both together:} bad product was likely made and the evidence was manipulated. The probability that subpotent or contaminated product reached patients is higher when both are present.
}

\vspace{8pt}
\begin{block}{Output}
\small 246 inspections $\times$ 17 features.\\
This matrix feeds the correlation analysis and the prediction model directly.
\end{block}

\vspace{4pt}
{\scriptsize Share and count are complementary: share is robust to long inspections; count captures absolute burden. Both enter the model.}
\end{columns}
\end{frame}

\section{Analysis}

\begin{frame}
\begin{center}

{\normalsize Do 483 text signals predict patient harm?\\ How well does a text-only model classify high-AE inspections?\\ What does the AE trajectory look like around an inspection?}
\end{center}

\begin{frame}{From 483 Text to Future Harm: The Detection Lag}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{The mechanism:}
\begin{itemize}
  \item FDA arrives. Product made before the visit is already in the market.
  \item The 483 records what was happening, not what starts next.
  \item AEs from pre-inspection product surface in the following quarters, before corrective action takes effect.
  \item We predict harm already in the pipeline, not future failures.
\end{itemize}

\column{0.45\textwidth}
\textbf{How we operationalize it:}

\vspace{6pt}
\begin{tabular}{ll}
\toprule
Element & Design \\
\midrule
Unit & One inspection event \\
Features & 483 text at that inspection \\
Outcome & AEs in Q+1..Q+4 (sum) \\
N & 246 inspections, 98 FEIs \\
CV & FEI-grouped 5-fold \\
\bottomrule
\end{tabular}

\vspace{8pt}
\textbf{Question:} which text signals predict AEs in the year after inspection, beyond knowing the OAI/VAI outcome?
\end{columns}
\end{frame}

% ── FAERS DETAIL SLIDE ────────────────────────────────────────────────────────

\begin{frame}{How We Attribute FAERS Reports to Manufacturers}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Serious outcomes only}\\[4pt]
We keep reports with at least one of: Death, Hospitalization, Life-threatening, Disability, Congenital anomaly, Required intervention, or Other serious outcome.\\
``No outcome reported'' excluded (14.6\% of reports).

\vspace{8pt}
\textbf{ANDA-level attribution}\\[4pt]
Each FAERS report carries \texttt{appl\_no} -- the application number for the specific manufacturer's product. We join on \texttt{appl\_no} to the Valisure NDC-ANDA-FEI crosswalk (built from Valisure's independently tested products).

\vspace{8pt}
\textbf{Why this matters}\\[4pt]
Drug-name joining assigns every atorvastatin AE to all 15 atorvastatin manufacturers. ANDA-level joining assigns each AE to the one manufacturer whose product the patient received.

\column{0.47\textwidth}
\begin{block}{Result}
144,286 serious FAERS reports $\to$ \textbf{68,500} matched to Valisure-tested ANDAs (47.5\%).\\[4pt]
Unmatched 52.5\% = manufacturers outside Valisure scope, not in analysis.\\[4pt]
Aggregated to quarterly counts per FEI.
\end{block}

% ── SECTION 6: PREDICTIVE MODEL ──────────────────────────────────────────────
\section{Predictive Model}

\begin{frame}{Prediction Setup}
\begin{columns}[T]
\column{0.52\textwidth}
\textbf{Unit:} One row per inspection.\\[4pt]
\textbf{Features:} 17 LLM text signals from the 483 at that inspection.\\[4pt]
\textbf{Outcome:} Were serious AEs in the year after inspection above the sample median?
$$\mathbf{1}\!\left[\sum_{q=1}^{4} \text{AE}_{t+q} > \text{median}\right]$$
{\small Pre-inspection product stays in the market for months. Summing Q+1 through Q+4 captures the full distribution window.}

\vspace{6pt}
\textbf{CV:} FEI-grouped 5-fold. All inspections of one facility stay in one fold.

\vspace{4pt}
\textbf{Model:} Logistic Regression (primary), Random Forest (check).

\column{0.45\textwidth}
\begin{tabular}{ll}
\toprule
Metric & Value \\
\midrule
Inspection events & 246 \\
Unique FEIs & 98 \\
Outcome base rate & 50\% \\
OAI inspections & 33 (13.4\%) \\
\bottomrule
\end{tabular}

\vspace{10pt}
\textbf{AUC = 0.5 is the floor.}\\[4pt]
{\small Outcome is 50/50 by construction. A random classifier scores exactly 0.5. What matters is whether text features beat that floor, and whether they beat the OAI label.}
\end{columns}
\end{frame}

\begin{frame}{483 Text Best Predicts Harm in Facilities FDA Never Penalized}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{AUC, 5-fold FEI-grouped CV (246 inspections, manufacturer-specific AEs):}

\vspace{4pt}
{\small
\begin{tabular}{clc}
\toprule
 & Feature set & LR AUC \\
\midrule
A & Text only (17 LLM features) & 0.503 \\
B & Text + OAI flag & 0.527 \\
C & OAI flag only & 0.539 \\
\midrule
\multicolumn{2}{l}{\textit{Random baseline}} & \textit{0.500} \\
\midrule
\highlight{D} & \highlight{Text, VAI-only facilities} & \highlight{0.586} \\
E & Text, OAI-ever facilities & 0.537 \\
\bottomrule
\end{tabular}
}

\vspace{6pt}
{\small Config D: 73 VAI-only FEIs, 165 inspections. These facilities never received a warning letter. The OAI flag has no variation -- text is the only predictor available.}

\column{0.47\textwidth}
{\small
\textbf{Full population (A vs C):} Text and the OAI flag perform similarly near random. Neither is a strong standalone classifier.

\vspace{5pt}
\textbf{VAI-only (D):} Text reaches \highlight{AUC\,=\,0.59} -- the strongest result. Among facilities FDA never formally penalized, 483 text is the only signal that separates high- and low-harm producers.

\vspace{5pt}
\textbf{OAI-ever (E):} Text is no better in facilities that did receive enforcement. The enforcement action itself may have resolved the quality problem, flattening future AEs.

\vspace{5pt}
\textbf{Takeaway:} 483 text is most valuable precisely where structured regulatory signals are absent.
}
\end{columns}
\end{frame}

% ── SECTION 10: SUMMARY ──────────────────────────────────────────────────────
\section{Summary}

\begin{frame}{Summary: Five Key Messages}

\begin{enumerate}
  \setlength{\itemsep}{6pt}
  \item \textbf{483 text predicts harm where structured signals are absent.}\\
    In facilities FDA never formally penalized (VAI-only), text-only LR AUC = \highlight{0.586}. In the full population, text and the OAI flag are near random and similar to each other.

  \item \textbf{Lab Controls -- not data integrity -- is the leading signal.}\\
    Four Lab Controls features reach significance at Q+4 ($\rho$ up to 0.28***). DI drops out with manufacturer-specific AE attribution -- it was an artifact of the drug-level join.

  \item \textbf{Inspection is reactive by design; 483 text is not.}\\
    FDA's Site Selection Model uses AEs and recalls to rank facilities. Rising AEs before inspection are partly expected. 483 text from the prior visit contains information the SSM does not incorporate.

  \item \textbf{VAI facilities with high text signals have elevated pre-inspection AEs.}\\
    Hi-sig VAI: manufacturer-specific AEs 25\% higher at inspection than one year prior. OAI-ever facilities show a flatter trajectory. FDA rated both the same.

  \item \textbf{Strategic leniency candidates are identifiable.}\\
    Facilities with repeated high Lab Controls signals and no OAI history can be ranked. Cross-referencing with market concentration will test the shortage-pressure hypothesis.
\end{enumerate}

\section{Limitations \& Next Steps}

\begin{frame}{Limitations}
\begin{columns}[T]
\column{0.50\textwidth}
\begin{itemize}
  \item \textbf{Sample size:} 98 FEIs, 33 OAI inspections. Subgroup results (VAI-only AUC\,=\,0.59) are hypothesis-generating, not confirmatory.
  \item \textbf{COVID gap:} 2020--21 moratorium left 4-year inter-inspection windows; post-2022 baselines are confounded.
  \item \textbf{FAERS coverage:} manufacturer-specific attribution matches 47.5\% of serious AEs. Unmatched reports are from manufacturers outside the Valisure scope.
\end{itemize}

\column{0.47\textwidth}
\begin{itemize}
  \item \textbf{Shared ANDAs:} one ANDA can cover multiple facilities. AE counts are proportional to market share, not individual facility output.
  \item \textbf{LLM circularity:} validated against Redica, which reads the same 483 text. Neither is inspector ground truth.
  \item \textbf{Volume:} SDUD covers Medicaid only ($\sim$20\% of market). Commercial volume not integrated.
  \item \textbf{Causal:} all associations are correlational. Strategic leniency interpretation needs market concentration data.
\end{itemize}
\end{columns}

\begin{frame}{Backup: FAERS Filtering — Design Rationale}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Why Primary Suspect only?}\\[4pt]
FAERS reports list all drugs a patient was taking. Only the PS was judged by the reporter to have caused the event. Including concomitant or interacting drugs would attribute AEs to bystander medications and inflate counts at every facility.

\vspace{8pt}
\textbf{Why first-word name match?}\\[4pt]
``Atorvastatin Calcium'' in \texttt{prod\_ai} has two words. Any-word matching recovers a handful of missed combination products but mis-attributes ``Calcium'' rows to Calcium Gluconate FEIs. First-word-only is the safer tradeoff; the unmatched 0.9\% is negligible.

\column{0.47\textwidth}
\textbf{Why not a rate (AEs / volume)?}\\[4pt]
SDUD Medicaid volume covers only 69\% of FEI-years. Dividing by a partially missing denominator introduces systematic bias — facilities with missing SDUD look artificially high-risk. AE count enters as the raw outcome; $\log(\text{SDUD units})$ enters as a covariate when available.

\vspace{8pt}
\textbf{Why serious outcomes only?}\\[4pt]
Non-serious reports are subject to much higher underreporting variability. Serious AEs have mandatory reporting paths (MedWatch, expedited), making the count more comparable across facilities and time periods.
\end{columns}
\end{frame}
